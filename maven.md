To use groovy-http with maven, include the following dependency:
```
    <dependencies>
        <!-- ... other dependency -->
        <dependency>
            <groupId>com.google.code</groupId>
            <artifactId>groovy-http</artifactId>
            <version>0.1</version>
        </dependency>
    </dependencies>

    <repositories>
        <!-- ... other repository -->
        <repository>
            <id>maven2-repository.dev.java.net</id>
            <name>Java.net Repository for Maven</name>
            <url>http://download.java.net/maven/2/</url>
        </repository>
    </repositories>
```
